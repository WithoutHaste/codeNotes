##JSON

making a class json-serializable
<code>
import json

class Target():
    def __init__(self):
        self.url = None
        self.title = None
        self.author = None
        
    def toJSON(self):
        return 'json.dumps(self, sort_keys=True, indent=4, default=lambda o: o.__dict__)
        
    @staticmethod
    def parse_json(self, json_str):
        json_obj = json.loads(json_str) #returns a dict
        target = Target()
        target.url = json_obj['url']
        target.title = json_obj['title']
        target.author = json_obj['author']
        return target
